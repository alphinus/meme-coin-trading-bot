    A hook for detecting first-time users and tracking onboarding completion via localStorage:

    ```typescript
    import { useState, useCallback } from "react";

    const ONBOARDING_KEY = "eluma-onboarding-completed";

    export function useOnboarding() {
      const [showOnboarding, setShow] = useState(() => {
        return !localStorage.getItem(ONBOARDING_KEY);
      });

      const completeOnboarding = useCallback(() => {
        localStorage.setItem(ONBOARDING_KEY, "true");
        setShow(false);
      }, []);

      const resetOnboarding = useCallback(() => {
        localStorage.removeItem(ONBOARDING_KEY);
        setShow(true);
      }, []);

      return { showOnboarding, completeOnboarding, resetOnboarding };
    }
    ```

    import { StepGetStarted } from "./StepGetStarted.js";

    interface OnboardingWizardProps {

    const STEPS: WizardStep[] = [
      { id: "welcome", labelKey: "wizard.onboarding.step_welcome", component: StepWelcome },
      { id: "flow", labelKey: "wizard.onboarding.step_flow", component: StepFlowOverview },
      { id: "start", labelKey: "wizard.onboarding.step_start", component: StepGetStarted },

    export function OnboardingWizard({ onDone }: OnboardingWizardProps) {
      function handleComplete(allData: Record<string, unknown>) {
        const action = (allData.action as string) || "explore";
        onDone(action);
      }

      return (
        <WizardShell
          steps={STEPS}
          onComplete={handleComplete}
          titleKey="wizard.onboarding.title"

      );
